766  void setRotation(const Vector& axis, float angleRadians) {
767  const Vector n = axis.normalized();
768  const float s = std::sin(angleRadians);
769  const float c = std::cos(angleRadians);
770  const float C = (1-c);
771 
772  xBasis = Vector(n[0]*n[0]*C + c, n[0]*n[1]*C - n[2]*s, n[0]*n[2]*C + n[1]*s);
773  yBasis = Vector(n[1]*n[0]*C + n[2]*s, n[1]*n[1]*C + c, n[1]*n[2]*C - n[0]*s);
774  zBasis = Vector(n[2]*n[0]*C - n[1]*s, n[2]*n[1]*C + n[0]*s, n[2]*n[2]*C + c );
775  }
